Trading Bot Integration

Connect your trading bot to this API. All endpoints are open — no auth required.

# Python example import requests BASE = "http://localhost:8000" # Search for items r = requests.get(f"{BASE}/api/v1/items/search?q=AK-47&source=steam") data = r.json() for item in data["items"]: print(f"{item['name']}: {item['price']} CNY") # Get bot arbitrage r = requests.get(f"{BASE}/api/v1/bot/arbitrage") for opp in r.json(): print(f"{opp['item_name']}: {opp['spread_pct']}% spread")
// JavaScript example const BASE = "http://localhost:8000"; // Search for items const res = await fetch(`${BASE}/api/v1/items/search?q=AK-47&source=steam`); const data = await res.json(); console.log(data.items.map(i => `${i.name}: ${i.price} CNY`)); // Trigger bot scan await fetch(`${BASE}/api/v1/bot/trigger-scan`, { method: "POST" });
# cURL example curl -s "http://localhost:8000/api/v1/items/search?q=AK-47&source=steam" | jq # Health check curl -s http://localhost:8000/api/v1/health | jq # Trigger bot scan curl -X POST http://localhost:8000/api/v1/bot/trigger-scan
